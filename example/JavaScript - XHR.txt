var data = new FormData();
data.append("filename", "sample.jpg");
data.append("url", "https://github.com/animemoeus/pyscord-storage/raw/master/sample.jpg");

var xhr = new XMLHttpRequest();
xhr.withCredentials = true;

xhr.addEventListener("readystatechange", function() {
  if(this.readyState === 4) {
    console.log(this.responseText);
  }
});

xhr.open("POST", "https://discord-storage.animemoe.us/upload-from-url/");

xhr.send(data);
