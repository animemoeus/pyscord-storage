var form = new FormData();
form.append("filename", "sample.jpg");
form.append("url", "https://github.com/animemoeus/pyscord-storage/raw/master/sample.jpg");

var settings = {
  "url": "https://discord-storage.animemoe.us/upload-from-url/",
  "method": "POST",
  "timeout": 0,
  "processData": false,
  "mimeType": "multipart/form-data",
  "contentType": false,
  "data": form
};

$.ajax(settings).done(function (response) {
  console.log(response);
});
