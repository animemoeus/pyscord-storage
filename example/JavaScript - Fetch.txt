var formdata = new FormData();
formdata.append("filename", "sample.jpg");
formdata.append("url", "https://github.com/animemoeus/pyscord-storage/raw/master/sample.jpg");

var requestOptions = {
  method: 'POST',
  body: formdata,
  redirect: 'follow'
};

fetch("https://discord-storage.animemoe.us/upload-from-url/", requestOptions)
  .then(response => response.text())
  .then(result => console.log(result))
  .catch(error => console.log('error', error));
